\section*{Summary}
Integration is the inverse process of differentiation. In the differential calculus, we are given a function and we have to find the derivative or differential of this function, but in the integral calculus, we are to find a function whose differential is given. 

Let $\frac{d}{dx}F(x) = f(x)$. Then we write $\int f(x) \, dx = F(x) + C$. These integrals are called indefinite integrals or general integrals, $C$ is called constant of integration. All these antiderivatives differ by as constant.

\subsection*{Properties of Indefinite Integrals:}
\begin{enumerate}
    \item $\int [f(x) + g(x)] \, dx = \int f(x) \, dx + \int g(x) \, dx$
    \item For any real number $k$, $\int k f(x) \, dx = k \int f(x) \, dx$
    \item $\int [k_1 f_1(x) + k_2 f_2(x) + \dots + k_n f_n(x)] \, dx = k_1 \int f_1(x) \, dx + \dots + k_n \int f_n(x) \, dx$
\end{enumerate}

\subsection*{Standard Integrals:}
\begin{enumerate}
    \item $\int x^n \, dx = \frac{x^{n+1}}{n+1} + C, (n \neq -1)$ 
    \item $\int \frac{1}{x} \, dx = \log |x| + C$
    \item $\int \cos x \, dx = \sin x + C$ 
    \item $\int \sin x \, dx = -\cos x + C$
    \item $\int \sec^2 x \, dx = \tan x + C$ 
    \item $\int \text{cosec}^2 x \, dx = -\cot x + C$
    \item $\int \sec x \tan x \, dx = \sec x + C$ 
    \item $\int \text{cosec} x \cot x \, dx = -\text{cosec} x + C$
    \item $\int \frac{dx}{\sqrt{1-x^2}} = \sin^{-1} x + C$ 
    \item $\int \frac{dx}{1+x^2} = \tan^{-1} x + C$
    \item $\int e^x \, dx = e^x + C$ 
    \item $\int a^x \, dx = \frac{a^x}{\log a} + C$
\end{enumerate}
\end{tcolorbox}

\subsection*{Integration by Partial Fractions:}
\begin{enumerate}
    \item $\frac{px+q}{(x-a)(x-b)} = \frac{A}{x-a} + \frac{B}{x-b}, a \neq b$
    \item $\frac{px+q}{(x-a)^2} = \frac{A}{x-a} + \frac{B}{(x-a)^2}$
    \item $\frac{px^2+qx+r}{(x-a)(x-b)(x-c)} = \frac{A}{x-a} + \frac{B}{x-b} + \frac{C}{x-c}$
    \item $\frac{px^2+qx+r}{(x-a)^2(x-b)} = \frac{A}{x-a} + \frac{B}{(x-a)^2} + \frac{C}{x-b}$
    \item $\frac{px^2+qx+r}{(x-a)(x^2+bx+c)} = \frac{A}{x-a} + \frac{Bx+C}{x^2+bx+c}$
\end{enumerate}

\subsection*{Integrals of Special Functions:}
\begin{enumerate}
    \item $\int \frac{dx}{x^2 - a^2} = \frac{1}{2a} \log \left| \frac{x-a}{x+a} \right| + C$
    \item $\int \frac{dx}{a^2 - x^2} = \frac{1}{2a} \log \left| \frac{a+x}{a-x} \right| + C$
    \item $\int \frac{dx}{x^2 + a^2} = \frac{1}{a} \tan^{-1} \frac{x}{a} + C$
    \item $\int \frac{dx}{\sqrt{x^2 - a^2}} = \log \left| x + \sqrt{x^2 - a^2} \right| + C$
    \item $\int \frac{dx}{\sqrt{a^2 - x^2}} = \sin^{-1} \frac{x}{a} + C$
    \item $\int \frac{dx}{\sqrt{x^2 + a^2}} = \log \left| x + \sqrt{x^2 + a^2} \right| + C$
\end{enumerate}

\subsection*{Fundamental Theorem of Integral Calculus:}
\begin{enumerate}
    \item \textbf{First Fundamental Theorem:} Let $A(x) = \int_a^x f(t) \, dt$, then $A'(x) = f(x)$.
    \item \textbf{Second Fundamental Theorem:} $\int_a^b f(x) \, dx = [F(x)]_a^b = F(b) - F(a)$.
\end{enumerate}

\subsection*{Integration by parts:}
$\int f_1(x) \cdot f_2(x) \, dx = f_1(x) \int f_2(x) \, dx - \int \left[ f'_1(x) \cdot \int f_2(x) \, dx \right] \, dx$
\\ Also, $\int e^x [f(x) + f'(x)] \, dx = e^x f(x) + C$
\end{tcolorbox}
